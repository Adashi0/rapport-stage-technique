Le capteur SCT-013 fournit une tension analogique proportionnelle au

\begin{lstlisting}[language=Python, caption={Lecture du capteur SCT-013 et publication MQTT}, label={lst:capteur}]
import paho.mqtt.client as mqtt
import Adafruit_ADS1x15
import time, json, math

BROKER_IP   = "192.168.1.100"
TOPIC       = "energie/salle1/courant"
CALIBRATION = 30.0   # facteur selon le capteur

adc    = Adafruit_ADS1x15.ADS1115()
client = mqtt.Client(client_id="capteur_salle1")
client.connect(BROKER_IP, 1883)

def lire_courant_rms(nb_echantillons=1000):
    somme = 0
    for _ in range(nb_echantillons):
        brut = adc.read_adc(0, gain=1)
        tension = brut * (4.096 / 32767)
        somme  += tension ** 2
    return math.sqrt(somme / nb_echantillons) * CALIBRATION

while True:
    courant = lire_courant_rms()
    payload = json.dumps({
        "timestamp": time.time(),
        "courant_A": round(courant, 3),
        "puissance_W": round(courant * 220, 1)
    })
    client.publish(TOPIC, payload)
    print(f"[MQTT] PubliÃĐ : {payload}")
    time.sleep(5)
\end{lstlisting}
